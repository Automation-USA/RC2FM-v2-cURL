Completed code QA

Purpose:
Clears all FileMaker global variables for RingCentral API session

Parameters:
None

Important Notes:
-None

Allow User Abort [ Off ]
# 
# clear user phone numbers
Set Variable [ $startingValue ; Value: JSONGetElement ( $$rc.responseUserPhoneNums ; ".paging.pageStart" ) ] 
Set Variable [ $NumberCount ; Value: JSONGetElement ( $$rc.responseUserPhoneNums ; ".paging.totalElements" ) ] 
# 
Set Variable [ $LoopCounter ; Value: $startingValue ] 
Loop
	Set Variable [ $recordNum ; Value: ".records[" & $LoopCounter & "]" ] 
	# 
	If [ not IsEmpty ( JSONGetElement ( $$rc.responseUserPhoneNums ; $recordNum & ".vanityPattern" ) ) ] 
		Set Variable [ $theLabel ; Value: "vanityNumber_" & $LoopCounter ] 
	Else
		Set Variable [ $theLabel ; Value: JSONGetElement ( $$rc.responseUserPhoneNums ; $recordNum & ".usageType" ) ] 
	End If
	Set Variable [ $NameValuePair ; Value: Evaluate ( "Let ( [ $$" & $theLabel & " = " & "\"\"" & " ] ; \"\" )" ) ] 
	# 
	Set Variable [ $LoopCounter ; Value: $LoopCounter + 1 ] 
	Exit Loop If [ $LoopCounter > $NumberCount ] 
End Loop
# 
# 
# clear account extensions
Set Variable [ $startingValue ; Value: JSONGetElement ( $$rc.responseAccountExtensions ; ".paging.pageStart" ) ] 
Set Variable [ $ExtensionCount ; Value: JSONGetElement ( $$rc.responseAccountExtensions ; ".paging.totalElements" ) ] 
# 
Set Variable [ $LoopCounter ; Value: $startingValue ] 
Loop
	Set Variable [ $recordNum ; Value: ".records[" & $LoopCounter & "]" ] 
	# 
	Set Variable [ $theLabel ; Value: "extension"  & JSONGetElement ( $$rc.responseAccountExtensions ; $recordNum & ".extension.extensionNumber" ) ] 
	Set Variable [ $NameValuePair ; Value: Evaluate ( "Let ( [ $$" & $theLabel & " = " & "\"\"" & " ] ; \"\" )" ) ] 
	# 
	Set Variable [ $anID ; Value: JSONGetElement ( $$rc.responseAccountExtensions ; $recordNum & ".extension.id" ) ] 
	Set Variable [ $theLabel ; Value: "extension"  & JSONGetElement ( $$rc.responseAccountExtensions ; $recordNum & ".extension.extensionNumber" ) & "_id" ] 
	Set Variable [ $NameValuePair ; Value: Evaluate ( "Let ( [ $$" & $theLabel & " = " & "\"\"" & " ] ; \"\" )" ) ] 
	# 
	Set Variable [ $LoopCounter ; Value: $LoopCounter + 1 ] 
	Exit Loop If [ $LoopCounter > $ExtensionCount ] 
End Loop
